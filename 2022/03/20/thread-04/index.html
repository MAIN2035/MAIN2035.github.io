线程加载与线程组

2022/3/20 Java Java多线程编程

饿汉模式(立即加载)

  • 使用类时已创建对象
package cn.day.b8;

public class MyObject {
    private static MyObject myObject = new MyObject();//使用类时已创建对象
    public static MyObject getInstance(){
        return myObject;
    }
}
class Thread1 extends Thread {
    @Override
    public void run(){
        System.out.println(MyObject.getInstance().hashCode());
    }
}
class Run{
    public static void main(String[] args) {
        var t1 = new Thread1();
        var t2 = new Thread1();
        var t3 = new Thread1();
        t1.start();
        t2.start();
        t3.start();
    }
}

懒汉模式（延迟加载）

  • 在属性方法（get方法）中实例化对象
  • 双检查锁实现懒汉单例模式
package cn.day.b8;
public class MyObject {
    private  volatile static MyObject myObject ;// volatile增加了可见性，阻止了创建MyObject时的代码重排
    public static MyObject getInstance(){
        try {
            if (myObject != null){
            }else {
                Thread.sleep(3000);
                synchronized (MyObject.class){
                    if (myObject == null){
                        myObject = new MyObject();
                    }
                }
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        return myObject;
    }
}
class Thread1 extends Thread {
    @Override
    public void run(){
        System.out.println(MyObject.getInstance().hashCode());
    }
}
class Run{
    public static void main(String[] args) {
        var t1 = new Thread1();
        var t2 = new Thread1();
        var t3 = new Thread1();
        t1.start();
        t2.start();
        t3.start();
    }
}
  • 静态内部类实现懒汉模式
package cn.day.b8;

public class MyObject {
    /*静态代码块实现懒汉模式
        private static MyObject myObject = null;
        static{
            myObject = new MyObject();
        }
    */
    private static class MyObjectHandler {
        private volatile static MyObject myObject = new MyObject();
    }
    public static MyObject getInstance() {
        return MyObjectHandler.myObject;
    }
}
class Thread1 extends Thread {

线程组

  • 一级关联：组中有线程而无线程下的线程
package cn.day.b8;

public class MyThread {
    public static void main(String[] args) {
        ThreadA aRunnable = new ThreadA();
        ThreadB bRunnable = new ThreadB();
        ThreadGroup group = new ThreadGroup("一级关联组");//构造一个新线程组。
        Thread aThread = new Thread(group,aRunnable);//加入线程组
        Thread bThread = new Thread(group,bRunnable);
        aThread.start();
        bThread.start();
        System.out.println("线组中线程程数 "+ group.activeCount());//返回该线程组及其子组的数量的估计值
        System.out.println("线程组名 "+ group.getName());//返回该线程组的名称
        Thread.sleep(3000);
        group.interrupt();//停止此线程组中所有线程
        Thread.enumerate(group);//将当前线程的线程组整体复制到指定线程组
    }
}
class ThreadA extends Thread{
    @Override
    public void run(){
        try {
            while (!Thread.currentThread().isInterrupted()){
                System.out.println("ThreadName "+ Thread.currentThread().getName());
                Thread.sleep(3000);
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}
class ThreadB extends Thread{
    @Override
    public void run(){
        try {
            while (!Thread.currentThread().isInterrupted()){
                System.out.println("ThreadName "+ Thread.currentThread().getName());
                Thread.sleep(3000);
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}
